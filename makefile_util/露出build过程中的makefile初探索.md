###  [bochs/misc/Makefile](https://gitcode.net/crk/bochs/-/blob/faa84bb0e4ddb9e3d6455ca82b2763a0aa4453ba/bochs/misc/Makefile)

```Makefile
#/bochs/misc/Makefile
.cc.o:
	$(CXX) -c $(CPPFLAGS) $(CXXFLAGS) $< -o $@

all: make_cmos_image

make_cmos_image: make_cmos_image.o
	$(LIBTOOL) --mode=link --tag CXX $(CXX) -o $@ $(CXXFLAGS_CONSOLE) $(LDFLAGS) make_cmos_image.o

clean:
	rm -f  *.o
	rm -f  make_cmos_image

dist-clean: clean
	rm -f  Makefile
```

###  [bochs/memory/Makefile](https://gitcode.net/crk/bochs/-/blob/faa84bb0e4ddb9e3d6455ca82b2763a0aa4453ba/bochs/memory/Makefile)

all: libmemory.a

.cc.o:
	$(CXX) -c $(BX_INCDIRS) $(CPPFLAGS) $(CXXFLAGS) $< -o $@

libmemory.a: $(BX_OBJS)
	rm -f  libmemory.a
	ar rv $@ $(BX_OBJS)
	$(RANLIB) libmemory.a

$(BX_OBJS): $(BX_INCLUDES)

clean:
	rm -f  *.o
	rm -f  *.a

dist-clean: clean
	rm -f  Makefile
